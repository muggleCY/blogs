    手写promise逻辑🔥

    ORANGEY 2023/4/23 JavaScript手写

    promise干什么的，解决什么问题？
    1.回调写起来不优雅（难以维护）回调地狱
    2.错误处理无法统一 我们需要处理公共错误逻辑（catch）
    3.尽量简化回调 解决多个并发问题（promise.all Promsie.finally）

    # Promise结构

    const p1 = new Promise((resolve, reject) => {
    console.log('Executor中的输出')
    resolve('result')
})

    上述代码执行可以看出:

    1. promise是一个构造函数 默认需要传入一个executor执行器
    2. executor会立刻执行(同步执行) 并且传入resolve 和 reject 两个参数
    3. promise 有三个状态 fulfilled rejected pending
    4. 可以通过resolve 和 reject来改变状态 同时调用相应的回调, promise实例状态变化后 不能再发生变化
    5. 当executor发生异常的时候 也会触发promise的失败

    根据上述特性可以编写代码

    const PENDING = 'PENDING'
const FULFILLED = 'FULFILLED'
const REJECTED = 'REJECTED'

// 定义MyPromise 这个类
class MyPromise {
    // 构造函数接收一个回调
    constructor(executor) {
        // 初始状态为pending
        this.status = PENDING
        // 初始化成功的值
        this.value = undefined
        // 初始化失败的原因
        this.reason = undefined
        // 成功的回调
        const resolve = (value) => {
            if (this.status === PENDING) {
                this.status = FULFILLED
                this.value = value
            }
        }
        // 失败的回调
        const reject = (reason) => {
            if (this.status === PENDING) {
                this.status = REJECTED
                this.reason = reason
            }
        }
        try {
            // 执行器
            executor(resolve, reject)
        } catch (error) {
            // 如果执行executor函数出错，直接执行reject
            reject(error)
        }
    }
}

    # Promise的then方法

    const p1 = new Promise((resolve, reject) => {
    console.log('Executor中的输出')
    setTimeout(() => {
        resolve('result')
    }, 1000)
})
p1.then(
    (res) => console.log(res),
    (err) => console.log(err)
)

    Promise中有一个then方法,里面有两个参数onFulfilled(成功的回调)和onRejected(失败的回调)

    1. 当状态为fullfilled时,将value值传入到onFulfilled执行; 当状态为rejected时,将value值传入到onRejected执行
    2. 由于onFulfilled和和onRejected要等this.value和this.reason有值了再执行，所以resolve函数和reject函数内部应该异步执行
    3. 由于promise实例在多处都会使用，所以可以通过数组对onFulfilled和和onRejected进行保存
    const PENDING = 'PENDING'
const FULFILLED = 'FULFILLED'
const REJECTED = 'REJECTED'

// 定义MyPromise 这个类
class MyPromise {
    // 构造函数接收一个回调
    constructor(executor) {
        // 初始状态为pending
        this.status = PENDING
        // 初始化成功的值
        this.value = undefined
        // 初始化失败的原因
        this.reason = undefined
        // 对onFulfilled和和onRejected进行保存的数组 初始化
        this.onResolvedCallbacks = []
        this.onRejectedCallbacks = []
        // 成功的回调
        const resolve = (value) => {
            if (this.status === PENDING) {
                this.status = FULFILLED
                this.value = value
                // 先保存 再挨个执行 微任务
                this.onResolvedCallbacks.forEach((fn) => fn())
            }
        }
        // 失败的回调
        const reject = (reason) => {
            if (this.status === PENDING) {
                this.status = REJECTED
                this.reason = reason
                // 先保存 再挨个执行 微任务
                this.onRejectedCallbacks.forEach((fn) => fn())
            }
        }
        try {
            // 执行器
            executor(resolve, reject)
        } catch (error) {
            // 如果执行executor函数出错，直接执行reject
            reject(error)
        }
    }

    then(onFulfilled, onRejected) {
        this.onResolvedCallbacks.push(onFulfilled)
        this.onRejectedCallbacks.push(onRejected)
    }
}

    问题:在开发过程中经常用到new Promise().then().then()
    then链的特点是成功或者失败的回调函数返回一个promise 内部会解析这个promise并且将结果传递到下一个then中

    const PENDING = 'PENDING'
const FULFILLED = 'FULFILLED'
const REJECTED = 'REJECTED'

// 定义MyPromise 这个类
class MyPromise {
    // 构造函数接收一个回调
    constructor(executor) {
        // 初始状态为pending
        this.status = PENDING
        // 初始化成功的值
        this.value = undefined
        // 初始化失败的原因
        this.reason = undefined
        // 对onFulfilled和和onRejected进行保存的数组 初始化
        this.onResolvedCallbacks = []
        this.onRejectedCallbacks = []
        // 成功的回调
        const resolve = (value) => {
            if (this.status === PENDING) {
                this.status = FULFILLED
                this.value = value
                // 先保存 再挨个执行 微任务
                this.onResolvedCallbacks.forEach((fn) => fn())
            }
        }
        // 失败的回调
        const reject = (reason) => {
            if (this.status === PENDING) {
                this.status = REJECTED
                this.reason = reason
                // 先保存 再挨个执行 微任务
                this.onRejectedCallbacks.forEach((fn) => fn())
            }
        }
        try {
            // 执行器
            executor(resolve, reject)
        } catch (error) {
            // 如果执行executor函数出错，直接执行reject
            reject(error)
        }
    }

    then(onFulfilled, onRejected) {
        return new MyPromise((resolve, reject) => {
            if (this.status === FULFILLED) {
                try {
                    const x = onFulfilled(this.value)
                    resolve(x)
                } catch (error) {
                    reject(error)
                }
            }
            if (this.status === REJECTED) {
                try {
                    const x = onRejected(this.reason)
                    reject(x)
                } catch (error) {
                    reject(error)
                }
            }
            if (this.status === PENDING) {
                this.onResolvedCallbacks.push(() => {
                    try {
                        const x = onFulfilled(this.value)
                        resolve(x)
                    } catch (error) {
                        reject(error)
                    }
                })
                this.onRejectedCallbacks.push(() => {
                    try {
                        const x = onRejected(this.reason)
                        reject(x)
                    } catch (error) {
                        reject(error)
                    }
                })
            }
        })
    }
}

    问题:此时发现三种情况的处理代码是高度类似的 而且对回调的值 有多种处理情况

    1. x 是个普通值(原始类型)
    2. x 是基于 Promises/A+ 规范 promise 对象
    3. x 是基于其他规范的 promise 对象
    4. x 就是 promise2

    所以将逻辑抽离出来成resolvePromise方法

    const PENDING = 'PENDING'
const FULFILLED = 'FULFILLED'
const REJECTED = 'REJECTED'

const resolvePromise = (promise2, x, resolve, reject) => {
  // 如果 promise 和 x 引用的同一对象
  if (promise2 === x) {
    return reject(new TypeError('Chaining cycle detected for promise'))
  }
  let called = false
  // x不是null 且x是对象或者函数
  if (x != null && (typeof x === 'object' || typeof x === 'function')) {
    try {
      let then = x.then
      // 如果then是函数，就默认是promise了
      if (typeof then === 'function') {
        then.call(
          x,
          (y) => {
            // 添加锁，避免成功后执行失败
            if (called) return
            called = true
            resolvePromise(promise2, y, resolve, reject)
          },
          (error) => {
            if (called) return
            called = true
            reject(error)
          }
        )
      } else {
        resolve(x) // 直接成功即可
      }
    } catch (error) {
      // 也属于失败
      if (called) return
      called = true
      reject(error)
    }
  } else {
    // 既不是对象也不是函数，说明是普通值
    resolve(x)
  }
}
// 定义MyPromise 这个类
class MyPromise {
  // 构造函数接收一个回调
  constructor (executor) {
    // 初始状态为pending
    this.status = PENDING
    // 初始化成功的值
    this.value = undefined
    // 初始化失败的原因
    this.reason = undefined
    // 对onFulfilled和和onRejected进行保存的数组 初始化
    this.onResolvedCallbacks = []
    this.onRejectedCallbacks = []
    // 成功的回调
    const resolve = (value) => {
      if (this.status === PENDING) {
        this.status = FULFILLED
        this.value = value
        // 先保存 再挨个执行 微任务
        this.onResolvedCallbacks.forEach((fn) => fn())
      }
    }
    // 失败的回调
    const reject = (reason) => {
      if (this.status === PENDING) {
        this.status = REJECTED
        this.reason = reason
        // 先保存 再挨个执行 微任务
        this.onRejectedCallbacks.forEach((fn) => fn())
      }
    }
    try {
      // 执行器
      executor(resolve, reject)
    } catch (error) {
      // 如果执行executor函数出错，直接执行reject
      reject(error)
    }
  }
  then (onFulfilled, onRejected) {
    onFulfilled =
      typeof onFulfilled === 'function' ? onFulfilled : (value) => value
    onRejected =
      typeof onRejected === 'function'
        ? onRejected
        : (err) => {
          throw err
        }
    const promise2 = new MyPromise((resolve, reject) => {
      if (this.status === FULFILLED) {
        setTimeout(() => {
          try {
            const x = onFulfilled(this.value)
            resolvePromise(promise2, x, resolve, reject)
          } catch (error) {
            reject(error)
          }
        })
      }
      if (this.status === REJECTED) {
        setTimeout(() => {
          try {
            const x = onRejected(this.reason)
            resolvePromise(promise2, x, resolve, reject)
          } catch (error) {
            reject(error)
          }
        })
      }
      if (this.status === PENDING) {
        this.onResolvedCallbacks.push(() => {
          setTimeout(() => {
            try {
              const x = onFulfilled(this.value)
              resolvePromise(promise2, x, resolve, reject)
            } catch (error) {
              reject(error)
            }
          })
        })
        this.onRejectedCallbacks.push(() => {
          setTimeout(() => {
            try {
              const x = onRejected(this.reason)
              resolvePromise(promise2, x, resolve, reject)
            } catch (error) {
              reject(error)
            }
          })
        })
      }
    })
    return promise2
  }
}

    # Promise.prototype.catch()

    返回一个Promise，并且处理拒绝的情况

    // catch方法其实就是执行一下then的第二个回调
catch (onRejected) {
    return this.then(undefined, onRejected)
}

    # Promise.prototype.finally()

    返回一个Promise。在promise结束时，无论结果是fulfilled或者是rejected，都会执行指定的回调函数。在finally之后，还可以继续then。并且会将值原封不动的传递给后面的then

    // finally()如果return了一个reject状态的Promise，
// 将会改变当前Promise的状态，
// 这个MyPromise.resolve就用于改变Promise状态，
finally (callback) {
    return this.then(
      (value) =>
        MyPromise.resolve(callback()).then(() => {
          return value
        }),
      (reason) =>
        MyPromise.resolve(callback()).then(() => {
          throw reason
        })
    )
}

    # Promise.resolve()

    返回一个以给定值解析后的Promise 对象

      static resolve (value) {
    if (value instanceof MyPromise) return value
    return new MyPromise((resolve) => resolve(value))
  }

    # Promise.reject()

    方法返回一个带有拒绝原因的Promise对象。

      static reject (reason) {
    return new MyPromise((resolve, reject) => reject(reason))
}

    # Promise.all()

    如果所有的promise都成功,返回数组--数组中包含没一个promise的返回结果;如果有一个失败则失败,返回的是第一个失败的结果

    static all (promiseArr) {
    // 定义一个计数器而不是不使用arr的lenght去判断 是为了避免普通值比较快导致前面的结果还是undefind而长度已经满足
    let index = 0
    let result = []
    return new MyPromise((resolve, reject) => {
      promiseArr.forEach((p, i) => {
        MyPromise.resolve(p).then(
          (val) => {
            index++
            result[i] = val
            if (index === promiseArr.length) {
              resolve(result)
            }
          },
          (error) => {
            // 有一个
            reject(error)
          }
        )
      })
    })
}

    # Promise.race()

    同时执行promise 有一个promise状态改变了 就返回

    static race (promiseArr) {
    return new MyPromise((resolve, reject) => {
      for (let i = 0; i < promiseArr.length; i++) {
        let curr = promiseArr[i]
        MyPromise.resolve(curr).then(value => resolve(value), error => reject(error))
      }
    })
}
